# 완전범죄

## **내 풀이**

```java
import java.util.*;

class Solution {
    public int solution(int[][] info, int n, int m) {
        int INF = 1_000_000;
        int[] dp = new int[m];
        Arrays.fill(dp, INF);
        dp[0] = 0;

        for (int[] item : info) {
            int[] next = new int[m];
            Arrays.fill(next, INF);

            for (int b = 0; b < m; b++) {
                if (dp[b] == INF) continue;

                next[b] = Math.min(next[b], dp[b] + item[0]);

                if (b + item[1] < m) {
                    next[b + item[1]] =
                        Math.min(next[b + item[1]], dp[b]);
                }
            }
            dp = next;
        }

        int answer = INF;
        for (int b = 0; b < m; b++) {
            if (dp[b] < n) {
                answer = Math.min(answer, dp[b]);
            }
        }

        return answer == INF ? -1 : answer;
    }
}

```
