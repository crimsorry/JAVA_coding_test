## 내 풀이

* hashmap 사용!

```python
import java.util.*;

class Solution {
    public int[] solution(String msg) {
        int index = 26;
        ArrayList<Integer> arr = new ArrayList<>();
        HashMap<String, Integer> hm = new HashMap<>();
        
        for(int i=0; i<index; i++){
            char c = (char) (i + 'A');
            hm.put(String.valueOf(c), i+1);
        }
        
        
        String s = msg.charAt(0) + "";
        
        for(int i=1; i<msg.length(); i++){
            String nextS = s + msg.charAt(i);
            if(hm.get(nextS) == null){
                arr.add(hm.get(s));
                hm.put(nextS, ++index);
                s = msg.charAt(i) + "";
            }else{
                s  = nextS; 
            }
        }
        
        if(s != null){
            arr.add(hm.get(s));
        }
        
        return arr.stream().mapToInt(a -> a).toArray();
    }
}
```
